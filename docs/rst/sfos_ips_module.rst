Examples
--------

.. code-block:: yaml+jinja

    - name: Enable IPS protection
      sophos.sophos_firewall.sfos_ips:
        username: "{{ username }}"
        password: "{{ password }}"
        hostname: "{{ inventory_hostname }}"
        port: 4444
        verify: false
        enabled: true
        state: updated
        delegate_to: localhost

    - name: Query IPS protection settings
      sophos.sophos_firewall.sfos_ips:
        username: "{{ username }}"
        password: "{{ password }}"
        hostname: "{{ inventory_hostname }}"
        port: 4444
        verify: false
        state: query
        delegate_to: localhost
