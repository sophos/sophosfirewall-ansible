Examples
--------

.. code-block:: yaml+jinja

    - name: Retrieve Service Group
      sophos.sophos_firewall.sfos_servicegroup:
        username: "{{ username }}"
        password: "{{ password }}"
        hostname: myfirewallhostname.sophos.net
        port: 4444
        verify: false
        name: TESTSERVICEGROUP
        state: query
      delegate_to: localservice

    - name: Create Service Group
      sophos.sophos_firewall.sfos_servicegroup:
        username: "{{ username }}"
        password: "{{ password }}"
        hostname: myfirewallhostname.sophos.net
        port: 4444
        verify: false
        name: TESTSERVICEGROUP
        description: Test Service Group
        service_list:
          - HTTP
          - HTTPS
        state: present
      delegate_to: localservice

    - name: Add Services to Service Group
      sophos.sophos_firewall.sfos_servicegroup:
        username: "{{ username }}"
        password: "{{ password }}"
        hostname: myfirewallhostname.sophos.net
        port: 4444
        verify: false
        name: TESTSERVICEGROUP
        description: Test Host Group
        service_list:
          - TESTSERVICE3
          - TESTSERVICE4
        action: add
        state: updated
      delegate_to: localservice
