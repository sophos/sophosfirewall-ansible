Examples
--------

.. code-block:: yaml+jinja

    - name: Retrieve Local service ACL exception rule
      sophos.sophos_firewall.sfos_service_acl_exception:
        username: "{{ username }}"
        password: "{{ password }}"
        hostname: myfirewallhostname.sophos.net
        port: 4444
        verify: false
        name: TESTACLRULE
        state: query

    - name: Create Local service ACL exception rule
      sophos.sophos_firewall.sfos_service_acl_exception:
        username: "{{ username }}"
        password: "{{ password }}"
        hostname: myfirewallhostname.sophos.net
        port: 4444
        verify: false
        name: TESTACLRULE
        description: Test ACL Rule
        position: bottom
        source_zone: LAN
        source_list:
          - TESTHOST1
          - TESTHOST2
        dest_list:
          - TESTHOST3
        service_list:
          - HTTP
          - HTTPS
        action: drop
        state: present
